311  void display()
312  {
313  //Declare the screen, to be updated before the final print statement.
314  char screen[mTermHeight][mTermWidth];
315 
316  //Clear the screen with ' ' characters.
317  for(int i = 0; i < mTermHeight; ++i)
318  {
319  for(int j = 0; j < mTermWidth; ++j)
320  {
321  screen[i][j] = ' ';
322  }
323  }
324 
325  /*
326  These are lambdas that convert between the coordinate systems of the terminal & function.
327  */
328  auto pixelToX = [&](int val)->double{
329  return mMap(val, 0, mTermWidth-1, mXrange[0], mXrange[1]);
330  };
331  //Unused, here for completion.
332  /*auto pixelToY = [&](int val)->double{
333  return mMap(val, 0, mTermHeight-1, mYrange[0], mYrange[1]);
334  };*/
335  auto xToPixel = [&](double val)->int{
336  return int(mMap(val, mXrange[0], mXrange[1], 0, mTermWidth));
337  };
338  auto yToPixel = [&](double val)->int{
339  return int(mMap(val, mYrange[0], mYrange[1], mTermHeight, 0));
340  };
341  /*
342  ------------
343  */
344 
345  //Get the positions of the axis, and make sure they're in-bounds (or don't render them)
346  int xaxis = yToPixel(0);
347  int yaxis = xToPixel(0);
348  /*
349  xaxis is the y-position of the xaxis, and vice-versa, which is why these comparisons seem off.
350  Here we are just filling in the axis's with their respective character.
351  */
352  if(yaxis >= 0 && yaxis < mTermWidth)
353  {
354  for(int i = 0; i < mTermHeight; ++i)
355  {
356  screen[i][yaxis] = '|';
357  }
358  }
359  if(xaxis >= 0 && xaxis < mTermHeight)
360  {
361  for(int i = 0; i < mTermWidth; ++i)
362  {
363  screen[xaxis][i] = '-';
364  }
365  }
366  /*
367  --------------
368  */
369 
370  //Update the screen w/ the functions.
371  for(auto &pair : mFunctions)
372  {
373  /*
374  For every function we're going to:
375  * Go through each X value (Term coords).
376  * Get the respective function-coordinate x position
377  * Pass it through the function.
378  * Assert that the y-value is not out of range.
379  * Convert back to term coords.
380  * If the pixel is too far from the previous pixel (and it's not the first execute),
381  * Iterate through from the last pixel to the current pixel's y-coordinate.
382  * Place a character at all points passed.
383  This generates smooth lines.
384  * Update the last pixel.
385  * Set the pixel at the actual point.
386  */
387 
388  //Set to max to avoid the first point being compared with whatever other initial value this may take on.
389  //(Try setting this to 0 and rendering x cubed.)
390  int last_pixel_y = INT32_MAX;
391 
392  //Iterate through all x-values (term coords).
393  for(int i = 0; i < mTermWidth; ++i)
394  {
395  //Get the corresponding x-value in function-coordinates.
396  double val = pixelToX(i);
397  //Pass it through the function to get the output (y-value).
398  double result = pair.first(val);
399 
400  //If the result is out of range, ignore and move on.
401  if(result < mYrange[0] || result > mYrange[1])
402  {
403  continue;
404  }
405 
406  //Otherwise, we're going to convert the point back to "pixel" coordinates.
407  int pixel_y = yToPixel(result);
408 
409  //Checks if the last pixel is too far from the new pixel,
410  //AND that it's not the first iteration that something was drawn (The INT32_MAX check).
411  if((last_pixel_y > pixel_y || last_pixel_y < pixel_y) && last_pixel_y != INT32_MAX)
412  {
413  //For every pixel in between..
414  for(int j = last_pixel_y;
415  j != pixel_y;
416  //Inc or dec based on which will terminate the loop >w<.
417  j += (last_pixel_y > pixel_y) ? (-1) : (1))
418  {
419  //Set that pixel.
420  screen[j][i] = pair.second;
421  }
422  }
423  //Set the las drawn pixel.
424  last_pixel_y = pixel_y;
425 
426  //And finally, actually, draw the real, actual pixel that was supposed to be drawn.
427  screen[pixel_y][i] = pair.second;
428  }
429  }
430 
431  //Render everything in one std::cout call.
432  std::stringstream ss;
433  for(int i = 0; i < mTermHeight; ++i)
434  {
435  for(int j = 0; j < mTermWidth; ++j)
436  {
437  //Push every character into a string stream...
438  ss << screen[i][j];
439  }
440  //..append newlines as needed...
441  ss << '\n';
442  }
443  //..and print!
444  std::cout << ss.str();
445  }
446  private:
447  std::vector<std::pair<Func, char>> mFunctions;
448  int mTermWidth;
449  int mTermHeight;

451  double mYrange[2];
